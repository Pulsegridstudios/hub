## Overview

The Community Outreach role is responsible for helping represent and promote Looped Operations across public-facing spaces.

This may include:

- Supporting community engagement  
- Assisting with partnerships or outreach  
- Helping communicate updates and announcements  
- Representing the project professionally in external communities  
- Encouraging growth while maintaining project standards  

Applicants should be confident communicating clearly, acting professionally, and representing the project in a positive and structured way.

---

## Eligibility Requirements

Applicants must:

- Be **16 years of age or older**  
- Demonstrate a **good level of written English communication**  
- Be able to represent the project professionally  
- Show good judgement when speaking on behalf of the project  
- Be comfortable engaging with community members and external audiences  
- Be willing to follow project standards and communication guidance  

> ⚠️ **Warning**  
> Applications from individuals under the age of 16 will not be considered.

---

## Expectations

Community Outreach contributors are expected to:

- Communicate clearly and professionally  
- Avoid making promises or statements without approval  
- Represent the project accurately  
- Escalate sensitive or uncertain matters where necessary  
- Maintain a positive and constructive public presence  

> ℹ️ **Note**  
> Public-facing communication may be reviewed before being used officially.

---

## Voluntary Roles

All roles within PulseGrid Studios are currently offered on a **voluntary basis**.

This means:

- No financial compensation is provided  
- Contributions are made at your own discretion  
- Experience gained may support personal development  

> ℹ️ **Note**  
> These roles are intended for individuals who wish to contribute to the project, gain experience, and be part of the development process.

<script>
document.addEventListener("DOMContentLoaded", function () {
  const form = document.getElementById("outreach-app-form");
  if (!form) return;

  const steps = Array.from(form.querySelectorAll(".form-step"));
  const prevBtn = document.getElementById("form-prev");
  const nextBtn = document.getElementById("form-next");
  const submitBtn = document.getElementById("outreach-app-submit");
  const indicator = document.getElementById("form-step-indicator");
  const status = document.getElementById("outreach-app-status");

  let currentStep = 0;

  function showStep(index) {
    steps.forEach((step, i) => {
      step.classList.toggle("is-active", i === index);
    });

    prevBtn.disabled = index === 0;
    indicator.textContent = `Step ${index + 1} of ${steps.length}`;

    const isLast = index === steps.length - 1;
    nextBtn.style.display = isLast ? "none" : "inline-flex";
    submitBtn.style.display = isLast ? "inline-flex" : "none";
  }

  function validateStep(index) {
    const fields = steps[index].querySelectorAll("input, textarea, select");
    for (const field of fields) {
      if (!field.checkValidity()) {
        field.reportValidity();
        return false;
      }
    }
    return true;
  }

  prevBtn.addEventListener("click", function () {
    if (currentStep > 0) {
      currentStep--;
      showStep(currentStep);
      form.scrollIntoView({ behavior: "smooth", block: "start" });
    }
  });

  nextBtn.addEventListener("click", function () {
    if (!validateStep(currentStep)) return;
    if (currentStep < steps.length - 1) {
      currentStep++;
      showStep(currentStep);
      form.scrollIntoView({ behavior: "smooth", block: "start" });
    }
  });

  form.addEventListener("submit", async function (event) {
    event.preventDefault();

    if (!validateStep(currentStep)) return;

    status.textContent = "";
    status.classList.remove("is-success", "is-error");
    submitBtn.disabled = true;
    submitBtn.textContent = "Submitting...";

    const formData = new FormData(form);
    const payload = {
      application_type: "community_outreach",
      roblox_name: String(formData.get("roblox_name") || "").trim(),
      discord_name: String(formData.get("discord_name") || "").trim(),
      timezone: String(formData.get("timezone") || "").trim(),
      community_time: String(formData.get("community_time") || "").trim(),
      community_experience: String(formData.get("community_experience") || "").trim(),
      team_experience: String(formData.get("team_experience") || "").trim(),
      experience_detail: String(formData.get("experience_detail") || "").trim(),
      preferred_area: String(formData.get("preferred_area") || "").trim(),
      why_outreach: String(formData.get("why_outreach") || "").trim(),
      good_fit: String(formData.get("good_fit") || "").trim(),
      professionalism: String(formData.get("professionalism") || "").trim(),
      scenario_question: String(formData.get("scenario_question") || "").trim(),
      scenario_conflict: String(formData.get("scenario_conflict") || "").trim(),
      scenario_announcement: String(formData.get("scenario_announcement") || "").trim(),
      agreement: formData.get("agreement") ? "Yes" : "No",
      website: String(formData.get("website") || "").trim()
    };

    try {
      const response = await fetch("https://lingering-frost-5384.frazergrant345.workers.dev/community-outreach-application", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify(payload)
      });

      const result = await response.json();

      if (!response.ok) {
        throw new Error(result.error || "Submission failed.");
      }

      form.reset();
      currentStep = 0;
      showStep(currentStep);
      status.textContent = "Application submitted successfully.";
      status.classList.add("is-success");
    } catch (error) {
      status.textContent = error.message || "Something went wrong.";
      status.classList.add("is-error");
    } finally {
      submitBtn.disabled = false;
      submitBtn.textContent = "Submit Application";
    }
  });

  showStep(currentStep);
});
</script>
